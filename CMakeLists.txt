PROJECT (CABLE)
INCLUDE_REGULAR_EXPRESSION ("^(cxx|cable|xml|gen|wrap|CxxTypes).*$")

SUBDIRS (Expat CxxTypes Parsers Generators Executables)
SUBDIR_DEPENDS(Parsers Expat CxxTypes)
SUBDIR_DEPENDS(Generators Parsers)
SUBDIR_DEPENDS(Executables Generators)

IF (CABLE_TCL_FACILITY_FORCE_ON)
  SET (CABLE_TCL_FACILITY ON)
ELSE (CABLE_TCL_FACILITY_FORCE_ON)
  IF (CABLE_TCL_FACILITY_FORCE_OFF)
    SET (CABLE_TCL_FACILITY OFF)
  ELSE (CABLE_TCL_FACILITY_FORCE_OFF)
    OPTION (CABLE_TCL_FACILITY "Build CABLE's Tcl wrapping facility." ON)
  ENDIF (CABLE_TCL_FACILITY_FORCE_OFF)
ENDIF (CABLE_TCL_FACILITY_FORCE_ON)

IF (CABLE_TCL_FACILITY)
  SUBDIRS(WrapTclFacility)
ENDIF (CABLE_TCL_FACILITY)

INCLUDE_DIRECTORIES (
${CABLE_SOURCE_DIR}
${CABLE_SOURCE_DIR}/Expat/xmlparse
)


CONFIGURE_FILE(${CABLE_SOURCE_DIR}/cable.README
               ${CABLE_BINARY_DIR}/cable.README.txt COPYONLY)
SET (CABLE_EXECUTABLE_OUTPUT_PATH ${EXECUTABLE_OUTPUT_PATH} CACHE INTERNAL "Single output directory for building all libraries.")
SET (CABLE_LIBRARY_OUTPUT_PATH ${LIBRARY_OUTPUT_PATH} CACHE INTERNAL "Single output directory for building all executables.")
